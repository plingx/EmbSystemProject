283  assert_param(IS_ETH_PAUSE_TIME(ETH_InitStruct->ETH_PauseTime));
284  assert_param(IS_ETH_ZEROQUANTA_PAUSE(ETH_InitStruct->ETH_ZeroQuantaPause));
285  assert_param(IS_ETH_PAUSE_LOW_THRESHOLD(ETH_InitStruct->ETH_PauseLowThreshold));
286  assert_param(IS_ETH_UNICAST_PAUSE_FRAME_DETECT(ETH_InitStruct->ETH_UnicastPauseFrameDetect));
287  assert_param(IS_ETH_RECEIVE_FLOWCONTROL(ETH_InitStruct->ETH_ReceiveFlowControl));
288  assert_param(IS_ETH_TRANSMIT_FLOWCONTROL(ETH_InitStruct->ETH_TransmitFlowControl));
289  assert_param(IS_ETH_VLAN_TAG_COMPARISON(ETH_InitStruct->ETH_VLANTagComparison));
290  assert_param(IS_ETH_VLAN_TAG_IDENTIFIER(ETH_InitStruct->ETH_VLANTagIdentifier));
291  /* DMA --------------------------*/
292  assert_param(IS_ETH_DROP_TCPIP_CHECKSUM_FRAME(ETH_InitStruct->ETH_DropTCPIPChecksumErrorFrame));
293  assert_param(IS_ETH_RECEIVE_STORE_FORWARD(ETH_InitStruct->ETH_ReceiveStoreForward));
294  assert_param(IS_ETH_FLUSH_RECEIVE_FRAME(ETH_InitStruct->ETH_FlushReceivedFrame));
295  assert_param(IS_ETH_TRANSMIT_STORE_FORWARD(ETH_InitStruct->ETH_TransmitStoreForward));
296  assert_param(IS_ETH_TRANSMIT_THRESHOLD_CONTROL(ETH_InitStruct->ETH_TransmitThresholdControl));
297  assert_param(IS_ETH_FORWARD_ERROR_FRAMES(ETH_InitStruct->ETH_ForwardErrorFrames));
298  assert_param(IS_ETH_FORWARD_UNDERSIZED_GOOD_FRAMES(ETH_InitStruct->ETH_ForwardUndersizedGoodFrames));
299  assert_param(IS_ETH_RECEIVE_THRESHOLD_CONTROL(ETH_InitStruct->ETH_ReceiveThresholdControl));
300  assert_param(IS_ETH_SECOND_FRAME_OPERATE(ETH_InitStruct->ETH_SecondFrameOperate));
301  assert_param(IS_ETH_ADDRESS_ALIGNED_BEATS(ETH_InitStruct->ETH_AddressAlignedBeats));
302  assert_param(IS_ETH_FIXED_BURST(ETH_InitStruct->ETH_FixedBurst));
303  assert_param(IS_ETH_RXDMA_BURST_LENGTH(ETH_InitStruct->ETH_RxDMABurstLength));
304  assert_param(IS_ETH_TXDMA_BURST_LENGTH(ETH_InitStruct->ETH_TxDMABurstLength));
305  assert_param(IS_ETH_DMA_DESC_SKIP_LENGTH(ETH_InitStruct->ETH_DescriptorSkipLength));
306  assert_param(IS_ETH_DMA_ARBITRATION_ROUNDROBIN_RXTX(ETH_InitStruct->ETH_DMAArbitration));
307  /*-------------------------------- MAC Config ------------------------------*/
308  /*---------------------- ETHERNET MACMIIAR Configuration -------------------*/
309  /* Get the ETHERNET MACMIIAR value */
310  tmpreg = ETH->MACMIIAR;
311  /* Clear CSR Clock Range CR[2:0] bits */
312  tmpreg &= MACMIIAR_CR_MASK;
313  /* Get hclk frequency value */
314  //RCC_GetClocksFreq(&rcc_clocks);
315  //hclk = rcc_clocks.HCLK_Frequency;
316  hclk=ETH_InitStruct->Sys_Clock_Freq*1000000; //得到系统时钟频率(单位:Hz)
317 
318  /* Set CR bits depending on hclk value */
319  if((hclk >= 20000000)&&(hclk < 35000000))
320  {
321  /* CSR Clock Range between 20-35 MHz */
322  tmpreg |= (uint32_t)ETH_MACMIIAR_CR_Div16;
323  }
324  else if((hclk >= 35000000)&&(hclk < 60000000))
325  {
326  /* CSR Clock Range between 35-60 MHz */
327  tmpreg |= (uint32_t)ETH_MACMIIAR_CR_Div26;
328  }
329  else if((hclk >= 60000000)&&(hclk < 100000000))
330  {
331  /* CSR Clock Range between 60-100 MHz */
332  tmpreg |= (uint32_t)ETH_MACMIIAR_CR_Div42;
333  }
334  else if((hclk >= 100000000)&&(hclk < 150000000))
335  {
336  /* CSR Clock Range between 100-150 MHz */
337  tmpreg |= (uint32_t)ETH_MACMIIAR_CR_Div62;
338  }
339  else /* ((hclk >= 150000000)&&(hclk <= 168000000)) */
340  {
341  /* CSR Clock Range between 150-168 MHz */
342  tmpreg |= (uint32_t)ETH_MACMIIAR_CR_Div102;
343  }
344 
345  /* Write to ETHERNET MAC MIIAR: Configure the ETHERNET CSR Clock Range */
346  ETH->MACMIIAR = (uint32_t)tmpreg;
347  /*-------------------- PHY initialization and configuration ----------------*/
348  /* Put the PHY in reset mode */
349  if(!(ETH_WritePHYRegister(PHYAddress, PHY_BCR, PHY_Reset)))
350  {
351  /* Return ERROR in case of write timeout */
352  return ETH_ERROR;
353  }
354 
355  /* Delay to assure PHY reset */
356  _eth_delay_(PHY_RESET_DELAY);
357 
358  if(ETH_InitStruct->ETH_AutoNegotiation != ETH_AutoNegotiation_Disable)
359  {
360  /* We wait for linked status... *///&
361 
362  do
363  {
364  timeout++;
365  } while (!(ETH_ReadPHYRegister(PHYAddress, PHY_BSR) ) && (timeout < PHY_READ_TO));
366 
367  /* Return ERROR in case of timeout */
368  if(timeout == PHY_READ_TO)
369  {
370  return ETH_ERROR;
371  }
372 
373  /* Reset Timeout counter */
374  timeout = 0;
375  /* Enable Auto-Negotiation */
376  if(!(ETH_WritePHYRegister(PHYAddress, PHY_BCR, PHY_AutoNegotiation)))
377  {
378  /* Return ERROR in case of write timeout */
379  return ETH_ERROR;
380  }
381 
382  /* Wait until the auto-negotiation will be completed *///& PHY_AutoNego_Complete)
383  do
384  {
385  timeout++;
386  }
387  while (!(ETH_ReadPHYRegister(PHYAddress, PHY_BSR)) && (timeout < (uint32_t)PHY_READ_TO));
388 
389  /* Return ERROR in case of timeout */
390  if(timeout == PHY_READ_TO)
391  {
392  return ETH_ERROR;
393  }
394 
395  /* Reset Timeout counter */
396  timeout = 0;
397 
398  /* Read the result of the auto-negotiation */
399  RegValue = ETH_ReadPHYRegister(PHYAddress, PHY_SR);
400 
401  /* Configure the MAC with the Duplex Mode fixed by the auto-negotiation process */
402  if((RegValue & PHY_DUPLEX_STATUS) != (uint32_t)RESET)
403  {
404  /* Set Ethernet duplex mode to Full-duplex following the auto-negotiation */
405  ETH_InitStruct->ETH_Mode = ETH_Mode_FullDuplex;
406  }
407  else
408  {
409  /* Set Ethernet duplex mode to Half-duplex following the auto-negotiation */
410  ETH_InitStruct->ETH_Mode = ETH_Mode_HalfDuplex;
411  }
412 
413  /* Configure the MAC with the speed fixed by the auto-negotiation process */
414  if(RegValue & PHY_SPEED_STATUS)
415  {
416  /* Set Ethernet speed to 10M following the auto-negotiation */
417  ETH_InitStruct->ETH_Speed = ETH_Speed_10M;
418  }
419  else
420  {
421  /* Set Ethernet speed to 100M following the auto-negotiation */
422  ETH_InitStruct->ETH_Speed = ETH_Speed_100M;
423  }
424  }
425  else
426  {
427  if(!ETH_WritePHYRegister(PHYAddress, PHY_BCR, ((uint16_t)(ETH_InitStruct->ETH_Mode >> 3) |
428  (uint16_t)(ETH_InitStruct->ETH_Speed >> 1))))
429  {
430  /* Return ERROR in case of write timeout */
431  return ETH_ERROR;
432  }
433  /* Delay to assure PHY configuration */
434  _eth_delay_(PHY_CONFIG_DELAY);
435 
436  }
437  /*------------------------ ETHERNET MACCR Configuration --------------------*/
438  /* Get the ETHERNET MACCR value */
439  tmpreg = ETH->MACCR;
440  /* Clear WD, PCE, PS, TE and RE bits */
441  tmpreg &= MACCR_CLEAR_MASK;
442  /* Set the WD bit according to ETH_Watchdog value */
443  /* Set the JD: bit according to ETH_Jabber value */
444  /* Set the IFG bit according to ETH_InterFrameGap value */
445  /* Set the DCRS bit according to ETH_CarrierSense value */
446  /* Set the FES bit according to ETH_Speed value */
447  /* Set the DO bit according to ETH_ReceiveOwn value */
448  /* Set the LM bit according to ETH_LoopbackMode value */
449  /* Set the DM bit according to ETH_Mode value */
450  /* Set the IPCO bit according to ETH_ChecksumOffload value */
451  /* Set the DR bit according to ETH_RetryTransmission value */
452  /* Set the ACS bit according to ETH_AutomaticPadCRCStrip value */
453  /* Set the BL bit according to ETH_BackOffLimit value */
454  /* Set the DC bit according to ETH_DeferralCheck value */

456  ETH_InitStruct->ETH_Jabber |

989 /**
990  * @brief This function should be called when a frame is received using DMA
991  * Receive interrupt, it allows scanning of Rx descriptors to get the
992  * the receive frame (should be used with interrupt mode only)
993  * @param None
994  * @retval Structure of type FrameTypeDef
995  */
996 FrameTypeDef ETH_Get_Received_Frame_interrupt(void)
997 {
998  FrameTypeDef frame={0,0,0};
999  __IO uint32_t descriptor_scan_counter = 0;
1000 
1001  /* scan descriptors owned by CPU */
1002  while (((DMARxDescToGet->Status & ETH_DMARxDesc_OWN) == (uint32_t)RESET)&&
1003  (descriptor_scan_counter<ETH_RXBUFNB))
1004  {
1005 
1006  /* Just by security */
1007  descriptor_scan_counter++;
1008 
1009  /* check if first segment in frame */
1010  if(((DMARxDescToGet->Status & ETH_DMARxDesc_FS) != (uint32_t)RESET)&&
1011  ((DMARxDescToGet->Status & ETH_DMARxDesc_LS) == (uint32_t)RESET))
1012  {
1013  DMA_RX_FRAME_infos->FS_Rx_Desc = DMARxDescToGet;
1014  DMA_RX_FRAME_infos->Seg_Count = 1;
1015  DMARxDescToGet = (ETH_DMADESCTypeDef*) (DMARxDescToGet->Buffer2NextDescAddr);
1016  }
1017 
1018  /* check if intermediate segment */
1019  else if (((DMARxDescToGet->Status & ETH_DMARxDesc_LS) == (uint32_t)RESET)&&
1020  ((DMARxDescToGet->Status & ETH_DMARxDesc_FS) == (uint32_t)RESET))
1021  {
1022  (DMA_RX_FRAME_infos->Seg_Count) ++;
1023  DMARxDescToGet = (ETH_DMADESCTypeDef*) (DMARxDescToGet->Buffer2NextDescAddr);
1024  }
1025 
1026  /* should be last segment */
1027  else
1028  {
1029  /* last segment */
1030  DMA_RX_FRAME_infos->LS_Rx_Desc = DMARxDescToGet;
1031 
1032  (DMA_RX_FRAME_infos->Seg_Count)++;
1033 
1034  /* first segment is last segment */
1035  if ((DMA_RX_FRAME_infos->Seg_Count)==1)
1036  DMA_RX_FRAME_infos->FS_Rx_Desc = DMARxDescToGet;
