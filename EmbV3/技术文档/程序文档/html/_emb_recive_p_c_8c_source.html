94 void ReadCmdPack(void)
95 {
96  u16 unCmdID,unCMDLength,unDataBufferSize;
97  u32 unTimestamp;
98  //判断帧头
99  if( ReadVERSION()!=1) return ;
100  //判断数据长度，若不够，则退出
101  if (tcp_server_buffer.tcp_recCount>CMD_HEAD_LENTH-2)
102  {
103  ReadDate(CMD_HEAD_LENTH-2);//读取信息
104  unCmdID=(unDataBuffer[1]<<8)+unDataBuffer[0];//命令unID
105  unTimestamp=unDataBuffer[2]+(unDataBuffer[3]<<8)+(unDataBuffer[4]<<16)+(unDataBuffer[5]<<24);//时间
106  unCMDLength=(unDataBuffer[7]<<8)+unDataBuffer[6];//帧长
107  }else return;
108  unDataBufferSize =unCMDLength-CMD_HEAD_LENTH;
109  //判断数据长度，若不够，则退出
110  if (tcp_server_buffer.tcp_recCount<unDataBufferSize) return;
111  //读指令数据，保护
112  if (unDataBufferSize<MAX_ATE_LENTH) ReadDate(unDataBufferSize); //读指令数据
113  else return;
114 
115  switch (unCmdID)
116  {
117  case ClearCMD_ID:
118  EmbClear(unDataBuffer,unDataBufferSize);break;//命令清空
119  /*手臂舵机*/
120  case ArmPositionMoveCMD_ID:
121  EmbArmMotorCmd(unDataBuffer,unDataBufferSize, unTimestamp,0);break;
122  case ArmSpeedMoveCMD_ID:
123  EmbArmMotorCmd(unDataBuffer,unDataBufferSize, unTimestamp,1);break;
124  case ArmPIDCMD_ID:
125  EmbArmPIDCmd(unDataBuffer,unDataBufferSize);break;
126  case ArmTorqueSwitchCMD_ID :
127  EmbArmTorqueSwitchCmd(unDataBuffer,unDataBufferSize);break;
128  case ArmMotorLimitCMD_ID:
129  EmbArmMotorLimitCmd(unDataBuffer,unDataBufferSize);break;
130  /*头部舵机*/
131  case HeadPositionMoveCMD_ID :
132  EmbHeadMotorCmd(unDataBuffer,unDataBufferSize, unTimestamp,0); break;
133  case HeadSpeedMoveCMD_ID :
134  EmbHeadMotorCmd(unDataBuffer,unDataBufferSize, unTimestamp,1); break;
135  case HeadPIDCMD_ID:
136  EmbHeadPIDCmd(unDataBuffer,unDataBufferSize);break;
137  case HeadTorqueSwitchCMD_ID :
138  EmbHeadTorqueSwitchCmd(unDataBuffer,unDataBufferSize);break;
139  case HeadMotorLimitCMD_ID:
140  EmbHeadMotorLimitCmd(unDataBuffer,unDataBufferSize);break;
141  /*轮子*/
142  case WheelPositionMoveCMD_ID:
143  EmbWheelCmd(unDataBuffer,unDataBufferSize, unTimestamp,0); break;
144  case WheelSpeedMoveCMD_ID:
145  EmbWheelCmd(unDataBuffer,unDataBufferSize, unTimestamp,1); break;
146  case WheelPIDCMD_ID:
147  EmbWheelPIDCmd(unDataBuffer,unDataBufferSize);break;
148  case WheelTorqueSwitchCMD_ID :
149  EmbWheelTorqueSwitchCmd(unDataBuffer,unDataBufferSize);break;
150  case WheelMotorLimitCMD_ID:

153  case ManipulationMoveCMD_ID:
