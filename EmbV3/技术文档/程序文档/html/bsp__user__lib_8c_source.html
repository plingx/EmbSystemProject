149 * 功能说明: 将整数转换为ASCII数组。支持负数。
150 * 形 参: _Number : 整数

156 void int_to_str(int _iNumber, char *_pBuf, unsigned char _len)
157 {
158  unsigned char i;
159  int iTemp;
160 
161  if (_iNumber < 0) /* 负数 */
162  {
163  iTemp = -_iNumber; /* 转为正数 */
164  }
165  else
166  {
167  iTemp = _iNumber;
168  }
169 
170  mem_set(_pBuf, ' ',_len);
171 
172  /* 将整数转换为ASCII字符串 */
173  for (i = 0; i < _len; i++)
174  {
175  _pBuf[_len - 1 - i] = (iTemp % 10) + '0';
176  iTemp = iTemp / 10;
177  if (iTemp == 0)
178  {
179  break;
180  }
181  }
182  _pBuf[_len] = 0;
183 
184  if (_iNumber < 0) /* 负数 */
185  {
186  for (i = 0; i < _len; i++)
187  {
188  if ((_pBuf[i] == ' ') && (_pBuf[i + 1] != ' '))
189  {
190  _pBuf[i] = '-';
191  break;
192  }
193  }
194  }
195 }

198 *********************************************************************************************************
199 * 函 数 名: str_to_int
200 * 功能说明: 将ASCII码字符串转换成整数。 遇到小数点自动越过。
201 * 形 参: _pStr :待转换的ASCII码串. 可以以逗号，#或0结束。 2014-06-20 修改为非0-9的字符。
202 * 返 回 值: 二进制整数值
203 *********************************************************************************************************
204 */
205 int str_to_int(char *_pStr)
206 {
207  unsigned char flag;
208  char *p;
209  int ulInt;
210  unsigned char i;
211  unsigned char ucTemp;
212 
213  p = _pStr;
214  if (*p == '-')
215  {
216  flag = 1; /* 负数 */
217  p++;
218  }
219  else
220  {
221  flag = 0;
222  }
223 
224  ulInt = 0;
225  for (i = 0; i < 15; i++)
226  {
227  ucTemp = *p;
228  if (ucTemp == '.') /* 遇到小数点，自动跳过1个字节 */
229  {
230  p++;
231  ucTemp = *p;
232  }
233  if ((ucTemp >= '0') && (ucTemp <= '9'))
234  {
235  ulInt = ulInt * 10 + (ucTemp - '0');
236  p++;
237  }
238  else
239  {
240  break;
241  }
242  }
243 
244  if (flag == 1)
245  {
246  return -ulInt;
247  }
248  return ulInt;
249 }

280 /*
281 *********************************************************************************************************
282 * 函 数 名: CRC16_Modbus
283 * 功能说明: 计算CRC。 用于Modbus协议。
284 * 形 参: _pBuf : 参与校验的数据
285 * _usLen : 数据长度
286 * 返 回 值: 16位整数值。 对于Modbus ，此结果高字节先传送，低字节后传送。
287 *
288 * 所有可能的CRC值都被预装在两个数组当中，当计算报文内容时可以简单的索引即可；
289 * 一个数组包含有16位CRC域的所有256个可能的高位字节，另一个数组含有低位字节的值；
290 * 这种索引访问CRC的方式提供了比对报文缓冲区的每一个新字符都计算新的CRC更快的方法；
291 *
292 * 注意：此程序内部执行高/低CRC字节的交换。此函数返回的是已经经过交换的CRC值；也就是说，该函数的返回值可以直接放置
293 * 于报文用于发送；
294 *********************************************************************************************************
295 */
296 uint16_t CRC16_Modbus(uint8_t *_pBuf, uint16_t _usLen)
297 {
298  uint8_t ucCRCHi = 0xFF; /* 高CRC字节初始化 */
299  uint8_t ucCRCLo = 0xFF; /* 低CRC 字节初始化 */
300  uint16_t usIndex; /* CRC循环中的索引 */
301 
302  while (_usLen--)
303  {
304  usIndex = ucCRCHi ^ *_pBuf++; /* 计算CRC */
305  ucCRCHi = ucCRCLo ^ s_CRCHi[usIndex];
306  ucCRCLo = s_CRCLo[usIndex];
307  }
308  return ((uint16_t)ucCRCHi << 8 | ucCRCLo);
309 }
