42 void HeadStart(u8 unStartID,u8 unNum)
43 {
44  u8 uni;
45  for (uni = unStartID; uni < unNum+unStartID; uni++)
46  {
47  //上一时刻位置速度，控制命令赋值
48  sMyServo[uni].sControl.nLastCmdPosition=
49  sHeadMotorCmdPack.sCmdBuffer[sHeadMotorCmdPack.unCmbRead][uni-Head_Start_ID].unGoalPosition;
50  sMyServo[uni].sControl.nLastCmdSpeed=0;
51  //上一时刻位置速度，下发命令赋值
52  sMyServo[uni].sControl.nLastExePosition=sMyServo[uni].sControl.nLastCmdPosition;
53  sMyServo[uni].sControl.nLastExeSpeed=0;
54  //当前时刻位置速度，控制命令赋值
55  sMyServo[uni].sControl.nCmdPosition=sMyServo[uni].sControl.nLastCmdPosition;
56  sMyServo[uni].sControl.nCmdSpeed=sMyServo[uni].sControl.nLastCmdSpeed;
57  if( 1==sHeadMotorCmdPack.unModeFlag) //速度模式
58  sMyServo[uni].sControl.nLastCmdPosition=
59  (sMyServo[uni].unPresentPositionH<<8)+sMyServo[uni].unPresentPositionL;
60  //判断是否为停止命令
61  if (ServoStopCMD==sHeadMotorCmdPack.sCmdBuffer[sHeadMotorCmdPack.unCmbRead][uni-Head_Start_ID].nSpeed)
62  sMyServo[uni].sControl.unStartFlag=0;
63  else{
64  sMyServo[uni].sControl.unStartFlag=1;
65  sMyServo[uni].sControl.unBrakeFlag=0;
66  sMyServo[uni].sControl.unStopFlag=0;
67  }
68  }
69  /*只有当条件满足时count--*/
70  if (sHeadMotorCmdPack.unCmbCount>1&&0==sHeadMotorCmdPack.unModeFlag&&Head_Servo_Num==unNum)
71  {
72  DISABLE_INT();
73  if (sHeadMotorCmdPack.unCmbRead!=sHeadMotorCmdPack.unCmbWrite&&sHeadMotorCmdPack.unCmbCount>0)
74  sHeadMotorCmdPack.unCmbRead++;
75  if (sHeadMotorCmdPack.unCmbRead >= ACTION_EXE_PACK_SIZE){
76  sHeadMotorCmdPack.unCmbRead = 0;}
77  sHeadMotorCmdPack.unCmbCount--;
78 
79  ENABLE_INT();
80  }
81  /*头部控制，开始标识*/
82  unHeadStartFlag=1;
83  unHeadMotorControlFlag=1;
84 }

106  sMyServo[uni].sControl.nLastCmdSpeed=sMyServo[uni].sControl.nCmdSpeed;
107  //新命令
108  sMyServo[uni].sControl.nCmdPosition=sHeadMotorCmdPack.sCmdBuffer[sHeadMotorCmdPack.unCmbRead][uni-Head_Start_ID].unGoalPosition;
109  //控制停止标志
110  if(0==sMyServo[uni].sControl.unStopFlag)
111  {
112  sMyServo[uni].sControl.unStopPos=sMyServo[uni].unPresentPositionL+(sMyServo[uni].unPresentPositionH<<8);
113  }
114  //检查是否为停止命令
115  if (ServoStopCMD==sHeadMotorCmdPack.sCmdBuffer[sHeadMotorCmdPack.unCmbRead][uni-Head_Start_ID].nSpeed)
116  {
117  //确认停止
118  if(1==sMyServo[uni].sControl.unStopFlag)
119  sMyServo[uni].unServoSendID=0xFE;
120  //刹车标识
121  sMyServo[uni].sControl.unBrakeFlag=1;
122  if(0==sHeadMotorCmdPack.unModeFlag)
123  sMyServo[uni].sControl.unStopFlag=1;
124  }else
125  {
126  sMyServo[uni].sControl.nCmdSpeed=sHeadMotorCmdPack.sCmdBuffer[sHeadMotorCmdPack.unCmbRead][uni-Head_Start_ID].nSpeed;
127  sMyServo[uni].sControl.unBrakeFlag=0;
128  sMyServo[uni].sControl.unStopFlag=0;
129  sMyServo[uni].sControl.unStartFlag=1;
130  sMyServo[uni].unServoSendID=uni;
131  }
132  }
133 }
