218 * 功能说明: 向串口发送一组数据。数据放到发送缓冲区后立即返回，由中断服务程序在后台完成发送
219 * 形 参: _ucPort: 端口号(COM1 - COM6)

221 * _usLen : 数据长度
222 * 返 回 值: 无
223 *********************************************************************************************************
224 */
225 void comSendBuf(COM_PORT_E _ucPort, uint8_t *_ucaBuf, uint16_t _usLen)
226 {
227  UART_T *pUart;
228 
229  pUart = ComToUart(_ucPort);
230 
231  if (pUart == 0)
232  {
233  return;
234  }
235 
236  if (pUart->SendBefor != 0)
237  {
238  pUart->SendBefor(); /* 如果是RS485通信，可以在这个函数中将RS485设置为发送模式 */
239  }
240 
241  UartSend(pUart, _ucaBuf, _usLen);
242 }
