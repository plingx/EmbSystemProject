35 /**
36 *********************************************************************************************************
37 * @name GetPowerStatus
38 * @brief 读电源开关板数据
39 * @details
40 *
41 * @param[in] _ucPort COM口ID
42 *
43 * @returns ERR_NONE Successfully
44 * Other values Unsuccessfully
45 * @author
46 *********************************************************************************************************
47 */
48 u8 GetPowerStatus(COM_PORT_E _ucPort)
49 {
50  u8 unLength=0,unID=0;
51  u16 uni,unj,unCheckCRC;
52  UART_T *pUart;
53  uint16_t usCount;
54  pUart = ComToUart(_ucPort);
55  if (NULL==pUart) return 0;
56  /* usRxWrite 变量在中断函数中被改写，主程序读取该变量时，必须进行临界区保护 */
57  DISABLE_INT();
58  usCount = pUart->usRxCount;
59  ENABLE_INT();
60  /* 如果读和写索引相同，则返回0 */
61  if (usCount<5) /* 小于6个数 */
62  {
63  return 0;
64  }
65  CopyUSARTRecieve(pUart,unBuf,usCount);/*COPY串口数据*/
66  for(unj=0;unj<usCount;unj++)
67  {
68  while(ReadHead(unBuf,&unj)) /*读头*/
69  {
70  if((unj+5)>usCount)//数量较少时回退
71  {
72  ReturnUSARTRecieve(pUart,usCount-unj);
73  return 1;
74  }
75  };
76  unID=unBuf[unj+2];
77  unLength=unBuf[unj+3];
78  if((unLength+unj)<=usCount&&((0x0d+8==unLength)||(0x1F+8==unLength)))
79  {
80  unCheckCRC=CRC16_Modbus(&unBuf[unj], unLength-2); //计算CRC校验位
81  if(unCheckCRC!=unBuf[unj-2+unLength]+(unBuf[unj-1+unLength]<<8))
82  unj+=1;
83  else
84  {
85  if(1==unHeadFlag)
86  s485WholeCount[20].un485RecieveCount++;
87  else
88  s485WholeCount[19].un485RecieveCount++;
89  switch(unBuf[unj+4])
90  {
91  case BatteryRead_H:
92  ReadPSMData(unBuf,&unj,unLength,unj);
93  break;
94  default:
95  break;
96  }
97  }
98  }else if(unLength+unj>usCount)
99  {
100  ReturnUSARTRecieve(pUart,usCount-unj);
101  return 1;
102  }
103  }
104  return 0;
105 }
106 /**
107 *********************************************************************************************************
108 * @name ReadHead
109 * @brief 读PSM帧头
110 * @details
111 *
112 * @param[in] *punBuf 数据buf
113 * @param[in] *punP 数组指针
114 *
115 * @returns 1 Successfully
116 * Other values Unsuccessfully
117 * @author
118 *********************************************************************************************************
119 */
120 u8 ReadHead(u8 *punBuf,u16 *punP)
121 {
122  u16 unHead;
123  u8 uni,unBuffer[2];
124  unHead=(punBuf[*punP+1]<<8)+punBuf[*punP];//头
125  unHeadFlag=0;
126  if (unHead!=BatteryHeader_H&&unHead!=PSMHeader_H) // 版本号判定，正确继续，不正退出
127  {
128  *punP+=1;
129  return 1;
130  }
131  if (BatteryHeader_H==unHead) //
132  {
133  unHeadFlag=1;
134  }
135  else
136  {
137  unHeadFlag=2;
138  }
139  return 0;
140 }
141 /**
142 *********************************************************************************************************
143 * @name ReadPSMData
144 * @brief 读取PSM数据
145 * @details
146 *
147 * @param[in] *punBuf 数据buf
148 * @param[in] *punP 数组指针
149 * @param[in] unLength 数据长度
150 * @param[in] unStartNum 数组起始指针数
151 *
152 * @returns ERR_NONE Successfully
153 * Other values Unsuccessfully
154 * @author
155 *********************************************************************************************************
156 */
157 u8 ReadPSMData(u8 *punBuf,u16 *punP,u16 unLength,u16 unStartNum)
158 {
159  u8 uni=0;
160  switch(unLength)
161  {
162  case 0x0D+8:
163 // s485WholeCount[20].un485RecieveCount++;
164  sBatteryCharger.unStatus=unBuf[unStartNum+6]+(unBuf[unStartNum+7]<<8);
165  sBatteryCharger.unExVoltage=unBuf[unStartNum+8]+(unBuf[unStartNum+9]<<8);
166  sBatteryCharger.unOutCurrent=unBuf[unStartNum+10]+(unBuf[unStartNum+11]<<8);
167  sBatteryCharger.unTemperature=unBuf[unStartNum+12];
168  sBatteryCharger.unBatteryVoltage=unBuf[unStartNum+13]+(unBuf[unStartNum+14]<<8);
169  sBatteryCharger.unBatteryInCurrent=unBuf[unStartNum+15]+(unBuf[unStartNum+16]<<8);
170  sBatteryCharger.unBatteryCapacity=unBuf[unStartNum+17]+(unBuf[unStartNum+18]<<8);
171 
172  *punP+=unLength-1;
173  break;
174  case 0x1F+8:
175 // s485WholeCount[19].un485RecieveCount++;
176  sEwayPSM.unRobotSwitchStatus=unBuf[unStartNum+6];
177  sEwayPSM.unRobotCurrentStatus=unBuf[unStartNum+7];
178  sEwayPSM.unPSM_PowerSwitchStatus=unBuf[unStartNum+8];
179  sEwayPSM.unPSM_Status1=unBuf[unStartNum+9];
180  sEwayPSM.unPSM_Status2=unBuf[unStartNum+10];
181  for(uni=0;uni<7;uni++)
182  {
183  sEwayPSM.sPSM_Out[uni].unVoltage=unBuf[unStartNum+13+2*uni]+(unBuf[unStartNum+14+2*uni]<<8);
184  }
185  for(uni=0;uni<5;uni++)
186  {
187  sEwayPSM.sPSM_Out[uni].unCurrent=unBuf[unStartNum+27+2*uni]+(unBuf[unStartNum+28+2*uni]<<8);
188  }
189  *punP+=unLength-1;
190  break;
191  default:
192  *punP+=1;
193  break;
194  }
195 }
