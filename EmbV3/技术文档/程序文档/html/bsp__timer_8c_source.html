148 void SysTick_ISR(void)
149 {
150  uint8_t i;
151  /* 每隔1ms进来1次 （仅用于 bsp_DelayMS） */
152  if (s_uiDelayCount > 0)
153  {
154  if (--s_uiDelayCount == 0)
155  {
156  s_ucTimeOutFlag = 1;
157  }
158  }
159  /* 每隔1ms，对软件定时器的计数器进行减一操作 */
160  for (i = 0; i < TMR_COUNT; i++)
161  {
162  bsp_SoftTimerDec(&s_tTmr[i]);
163  }
164 
165  /* 全局运行时间每1ms增1 */
166  g_iRunTime++;
167 
168  if (g_iRunTime == 0x7FFFFFFF) /* 这个变量是 int32_t 类型，最大数为 0x7FFFFFFF */
169  {
170  g_iRunTime = 0;
171  }
172  lwip_localtime++; //加1
173  if(lwip_localtime-localtime>=1000)
174  {
175  localtime=lwip_localtime;
176  LED0=!LED0;
177  }
178 
179  s30ms_count++;
180  if (s30ms_count >29)
181  {
182  s30ms_count = 0;
183  Task.Flag.All=0;
184  /* 每隔20ms调用一次此函数，此函数在 bsp.c */
185  }
186 }

208  _tmr->Count = _tmr->PreLoad;
209  }
210  }

215 *********************************************************************************************************
216 * 函 数 名: bsp_DelayMS
217 * 功能说明: ms级延迟，延迟精度为正负1ms
218 * 形 参: n : 延迟长度，单位1 ms。 n 应大于2
219 * 返 回 值: 无
220 *********************************************************************************************************
221 */
222 void bsp_DelayMS(uint32_t n)
223 {
224  if (n == 0)
225  {
226  return;
227  }
228  else if (n == 1)
229  {
230  n = 2;
231  }
232 
233  DISABLE_INT(); /* 关中断 */
234 
235  s_uiDelayCount = n;
236  s_ucTimeOutFlag = 0;
237 
238  ENABLE_INT(); /* 开中断 */
239 
240  while (1)
241  {
242  bsp_Idle(); /* CPU空闲执行的操作， 见 bsp.c 和 bsp.h 文件 */
243 
244  /*
245  等待延迟时间到
246  注意：编译器认为 s_ucTimeOutFlag = 0，所以可能优化错误，因此 s_ucTimeOutFlag 变量必须申明为 volatile
247  */
248  if (s_ucTimeOutFlag == 1)
249  {
250  break;
251  }
252  }
253 }

255 /*
256 *********************************************************************************************************
257 * 函 数 名: bsp_DelayUS
258 * 功能说明: us级延迟。 必须在systick定时器启动后才能调用此函数。
259 * 形 参: n : 延迟长度，单位1 us
260 * 返 回 值: 无
261 *********************************************************************************************************
262 */
263 void bsp_DelayUS(uint32_t n)
264 {
265  uint32_t ticks;
266  uint32_t told;
267  uint32_t tnow;
268  uint32_t tcnt = 0;
269  uint32_t reload;
270 
271  reload = SysTick->LOAD;
272  ticks = n * (SystemCoreClock / 1000000); /* 需要的节拍数 */
273 
274  tcnt = 0;
275  told = SysTick->VAL; /* 刚进入时的计数器值 */
276 
277  while (1)
278  {
279  tnow = SysTick->VAL;
280  if (tnow != told)
281  {
282  /* SYSTICK是一个递减的计数器 */
283  if (tnow < told)
284  {
285  tcnt += told - tnow;
286  }
287  /* 重新装载递减 */
288  else
289  {
290  tcnt += reload - tnow + told;
291  }
292  told = tnow;
293 
294  /* 时间超过/等于要延迟的时间,则退出 */
295  if (tcnt >= ticks)
296  {
297  break;
298  }
299  }
300  }
301 }
