60 /**
61 *********************************************************************************************************
62 * @name PSUInfoPack
63 * @brief 生成PSU 数据包
64 * @details
65 *

67 *
68 * @returns ERR_NONE Successfully
69 * Other values Unsuccessfully
70 * @author
71 *********************************************************************************************************
72 */
73 void PSUInfoPack(void)
74 {
75  u8 unBuffer[PSU_Status_Length];
76  u8 uni;
77  //包头sBatteryCharger
78  EmbInfoAddHead(unBuffer,PSU_Status,PSU_Status_Length);
79  /*充电控制工作板状1
80  bit0：过热，
81  bit1：输出过流（输出电流超过10A），
82  bit2:输入欠压，
83  bit3：输入过压，
84  bit4：电池欠压，
85  bit5：输出短路，
86  bit6：电池即将输出关闭，
87  bit7：未定义错误；以上0均代表无故障
88  */
89  unBuffer[10]=0xFF&sBatteryCharger.unStatus;
90  /*充电控制工作板状2
91  bit0：负载供电电源（0是外部电源，1是电池），
92  bit1：电池充满（1是充满），
93  bit6-bit7：00是未充电，01是充电恒流，10是充电恒压，11是充电涓流；其他未定义位补0
94  */
95  unBuffer[11]=0xFF&(sBatteryCharger.unStatus>>8);
96 
97  /*外部输入电压，数值放大了100倍，单位0.01V*/
98  unBuffer[12]=0xFF&sBatteryCharger.unExVoltage;
99  unBuffer[13]=0xFF&(sBatteryCharger.unExVoltage>>8);
100  /*负载电流，数值放大了100倍，单位0.01A*/
101  unBuffer[14]=0xFF&sBatteryCharger.unOutCurrent;
102  unBuffer[15]=0xFF&(sBatteryCharger.unOutCurrent>>8);
103  /*驱动板当前温度*/
104  unBuffer[16]=0xFF&sBatteryCharger.unTemperature;
105  /*电池电压，数值放大了100倍，单位0.01V*/
106  unBuffer[17]=0xFF&sBatteryCharger.unBatteryVoltage;
107  unBuffer[18]=0xFF&(sBatteryCharger.unBatteryVoltage>>8);
108  /*电池充电电流，数值放大了100倍，单位0.01A*/
109  unBuffer[19]=0xFF&sBatteryCharger.unBatteryInCurrent;
110  unBuffer[20]=0xFF&(sBatteryCharger.unBatteryInCurrent>>8);
111  /*电池当前容量，数值放大了100倍，单位0.01%*/
112  unBuffer[21]=0xFF&sBatteryCharger.unBatteryCapacity;
113  unBuffer[22]=0xFF&(sBatteryCharger.unBatteryCapacity>>8);
114  TcpSendBufferAddData(unBuffer,PSU_Status_Length);
115 }

128 void PSMInfoPack(void)
129 {
130  u8 unBuffer[PSM_Status_Length];
131  u8 uni;
132  //包头sEwayPSM
133  EmbInfoAddHead(unBuffer,PSM_Status,PSM_Status_Length);
134  //PowerLevel
135  /*电源开关板七路电压电流状态:
136  0 sPC;
137  1 sLeftArm_PSM;
138  2 sRightArm_PSM;
139  3 sWheel_PSM;
140  4 sElse_PSM;
141  5 sDC5V_PSM;
142  6 sDC12V_PSM;
143  */
144  for(uni=0;uni<7;uni++)
145  {
146  /*电压，数值放大了100倍，单位0.01V*/
147  unBuffer[uni*4+10]=0xFF&sEwayPSM.sPSM_Out[uni].unVoltage;
148  unBuffer[uni*4+11]=0xFF&(sEwayPSM.sPSM_Out[uni].unVoltage>>8);
149  /*电流，数值放大了100倍，单位0.01A*/
150  unBuffer[uni*4+12]=0xFF&sEwayPSM.sPSM_Out[uni].unCurrent;
151  unBuffer[uni*4+13]=0xFF&(sEwayPSM.sPSM_Out[uni].unCurrent>>8);
152 
153  }
154  uni=uni-1;
155  /*机器人当前开机状态：
156  0代表未开机状态，
157  1代表已开机正在启动，
158  2代表已开机，
159  3代表休眠状态，
160  4代表急停状态，
161  5等待关机状态
162  */
163  unBuffer[uni*4+14]=sEwayPSM.unRobotCurrentStatus;
164  /*各路开关状态:
165  bit0代表上位机供电回路,1代表闭合，0代表打开状态；
166  Bit1代表左臂；
167  Bit2代表底盘；
168  Bit3代表右臂；
169  Bit4代表其他；
170  Bit5代表DCDC；
171  Bit6代表上位机已开机，1：已开机
172  */
173  unBuffer[uni*4+15]=sEwayPSM.unPSM_PowerSwitchStatus;
174  /*电源开关板当前状态2:
175  bit1-0代表其他模块； 00代表正常，01代表欠压，10代表过流；
176  bit3-2代表DCDC5V回路；
177  bit5-4代表DCDC12V回路；
178  */
179  unBuffer[uni*4+16]=sEwayPSM.unPSM_Status1;
180  /*电源开关板当前状态2:
181  bit1-0代表其他模块； 00代表正常，01代表欠压，10代表过流；
182  bit3-2代表DCDC5V回路；
183  bit5-4代表DCDC12V回路；
184  */
185  unBuffer[uni*4+17]=sEwayPSM.unPSM_Status2;
186 
187  TcpSendBufferAddData(unBuffer,PSM_Status_Length);
188 }
