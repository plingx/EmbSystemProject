259 {
260  u8 unSendBuf[256],unSendLenth=0,uni;

273  unCRC=CRC16_Modbus(&unSendBuf[0],unSendLenth-2);//校验

275  unSendBuf[unSendLenth-1]=(unCRC>>8)&0XFF;

277  for(uni=0;uni<15;uni++)
278  {
279  sMyServo[uni].unPresentSpeedL=0x7531&0xff;
280  sMyServo[uni].unPresentSpeedH=(0x7531>>8)&0xff;

286  s485WholeCount[13].un485SendCount++;

292  s485WholeCount[3].un485SendCount++;
293  s485WholeCount[4].un485SendCount++;
294  s485WholeCount[5].un485SendCount++;
295  break;
296  case 4:

298  s485WholeCount[7].un485SendCount++;
299  s485WholeCount[8].un485SendCount++;
300  s485WholeCount[9].un485SendCount++;

306 }

326 {
327  u8 unSendBuf[256],unSendLenth=0,uni,unj;
