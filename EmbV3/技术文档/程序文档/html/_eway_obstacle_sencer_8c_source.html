76  for (uni=0;uni<8;uni++)
77  {
78  sObstacleSencer.unDistans[uni]=sonar_GetDistance(uni);//获取声呐信息

81  return 0;
82  sWheelMotorSpeed[0]= -sWheelMotor[0].nPosCtrlSpeedHZ/10.0/90.0*WheelB;//左轮:cm/秒
83  sWheelMotorSpeed[1]= sWheelMotor[1].nPosCtrlSpeedHZ/10.0/90.0*WheelB;
84  if (sWheelMotorSpeed[1]*sWheelMotorSpeed[0]<0)
85  return 0;
86  ObstacleSpeed=(abs(sWheelMotorSpeed[0])>abs(sWheelMotorSpeed[1])) ? sWheelMotorSpeed[0]:sWheelMotorSpeed[1];
87 
88  RealObstacleDis=ObstacleSpeed*ObstacleSpeed;///200.0/0.4;//厘米
89  RealObstacleDis=RealObstacleDis/200.0*2.5+15;//厘米
90  if (RealObstacleDis<=35)
91  RealObstacleDis=35;
92  if (ObstacleSpeed>0)//后退
93  {
94  if((sObstacleSencer.unDistans[1]<=RealObstacleDis)||(sObstacleSencer.unDistans[7]<=RealObstacleDis))
95  {
96  EmbWheelBrake(); /*刹车*/
97  unAvoidingBrakeFlag=1;
98  return 1;
99  }
100  }
101  else if (ObstacleSpeed!=0)//
102  {
103  if((sObstacleSencer.unDistans[0]<=RealObstacleDis))
104  {
105  EmbWheelBrake();/*刹车*/
106  unAvoidingBrakeFlag=1;
107  return 1;
108  }
109  for (uni=2;uni<5;uni++)
110  {
111  if(sObstacleSencer.unDistans[0]<=RealObstacleDis)
112  {
113  EmbWheelBrake();/*刹车*/
114  unAvoidingBrakeFlag=1;
115  return 1;
116  }
117  if(sObstacleSencer.unDistans[uni]<=RealObstacleDis)
118  {
119  EmbWheelBrake();/*刹车*/
120  unAvoidingBrakeFlag=1;
121  return 1;
122  }
123  }
124  }
