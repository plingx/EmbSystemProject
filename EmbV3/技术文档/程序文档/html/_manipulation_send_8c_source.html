42 void ManipulationSetPosition(void)
43 {
44  u8 unRes=0,unLength,unCount,uni=0,unStartID=0,unCheckSum = 0;
45  u16 unPos,unSpeed,unSum = 0;
46  u8 unBuf[5*2+9];
47  unCount=1;
48  //判断收到命令个数
49  if(1==sManipulationCmdPack.unCmdGetFlag[0]&&1==sManipulationCmdPack.unCmdGetFlag[1])
50  {
51  unRes=3;
52  unLength=14;unCount=2;
53  }
54  else if(1==sManipulationCmdPack.unCmdGetFlag[0])
55  {
56  unRes=2;unLength=9;
57  }
58  else if(1==sManipulationCmdPack.unCmdGetFlag[1])
59  {
60  unRes=1;unLength=9;unStartID=1;
61  }
62  else {sManipulationCmdPack.unCmbCount=0;return;}
63  unBuf[0]=0xFF; //hesd1
64  unBuf[1]=0xFF; //head2
65  unBuf[2]=0xFE; //Broadcasting ID 254
66  unBuf[3]=unLength; //datalength ((4+1)*10+4=54)
67  unBuf[4]=0x83; //instruction SYNC WRITE control several RX-64s simultaneously
68  unBuf[5]=0x1E; //position low regisiter addunRess
69  unBuf[6]=0x04; //every dynamixel position(2) and speed (2)length
70  for (uni=unStartID;uni<unCount;uni++)
71  {
72  unPos=sManipulationCmdPack.sCmdBuffer[0][uni].unGoalPosition;
73  unSpeed=sManipulationCmdPack.sCmdBuffer[0][uni].unSpeed;
74  unBuf[uni*5+7]= uni+1;
75  if(0==unSpeed)//若速度为零，不下发
76  unBuf[uni*5+7]=0xF0;
77  unBuf[uni*5+8]= unPos&0xFF;
78  unBuf[uni*5+9]= (unPos>>8)&0xFF;
79  unBuf[uni*5+10]= unSpeed&0xFF;
80  unBuf[uni*5+11]= (unSpeed>>8)&0xFF;
81  unSum = unSum + unBuf[uni*5+7]+unBuf[uni*5+8]+unBuf[uni*5+9]+unBuf[uni*5+10]+unBuf[uni*5+11];
82  }
83  unSum = unSum + 0x1A7 + 5*unCount; // 0xFE+5*SERVO_NUM+4+0x83+0x1E+0x04 = 0x1A7+5*SERVO_NUM
84  if(unSum > 0xFF)
85  {
86  unSum = unSum & 0x00FF ;
87  }
88  if (uni==2)
89  uni=uni-1;
90  unCheckSum = 0xFF - unSum;
91  unBuf[uni*5+12]=unCheckSum;
92  sManipulationCmdPack.unCmbCount=0;
93  sManipulationCmdPack.unCmdGetFlag[0]=0;
94  sManipulationCmdPack.unCmdGetFlag[1]=0;
95  comSendBuf(COM5,unBuf,uni*5+13);
96 }
