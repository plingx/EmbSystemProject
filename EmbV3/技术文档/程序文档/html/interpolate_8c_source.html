461 /**
462 *********************************************************************************************************
463 * @name InterpolateSpeedPosition
464 * @brief 低阶插值函数
465 * @details
466 *
467 * @param[in] nInitPosition 初始位置
468 * @param[in] fInitSpeed 初始速度
469 * @param[in] nEndPosition 末位置
470 * @param[in] fEndSpeed 末速度
471 * @param[in] nTempNum 插值次序0~3
472 *
473 * @returns ERR_NONE Successfully
474 * Other values Unsuccessfully
475 * @author
476 *********************************************************************************************************
477 */
478  static void InterpolateSpeedPosition(int nInitPosition, float fInitSpeed, int nEndPosition, float fEndSpeed, int nTempNum)
479 {
480  // in this function, speed is defined as: position diff per control cycle
481  float fGoalSpeed; int nGoalPosition; int nPosDiff;
482  float fParaA ;
483  float fParaB ;
484  nPosDiff = nEndPosition - nInitPosition;
485  if (nPosDiff==0)
486  {
487  nTempGoalPosition=nEndPosition;
488  fTempDiffPerCycle=0;
489  return;
490  }
491  //120ms
492  fParaA = (float)(2 * nPosDiff - 3 * fEndSpeed - 5 * fInitSpeed) / 8.0;
493  fParaB = (float)( fInitSpeed + 7 * fEndSpeed - 2*nPosDiff) / 8.0;
494  if(nTempNum <= 1){
495  fGoalSpeed = fInitSpeed + fParaA * (nTempNum + 1);
496  nGoalPosition = nInitPosition + fInitSpeed * (nTempNum + 1) + fParaA * (nTempNum + 1) * (nTempNum + 2) / 2; }
497  else{
498  fGoalSpeed = fInitSpeed + fParaA * 2 + fParaB * (nTempNum - 1);
499  nGoalPosition = nInitPosition + fInitSpeed*(nTempNum+1) + fParaA*(1+2*nTempNum) + fParaB*(nTempNum-1)*(nTempNum)/2;
500  }
501  nTempGoalPosition=nGoalPosition;
502  fTempDiffPerCycle=fGoalSpeed;
503  return;
504 }
