30 *********************************************************************************************************
31 * @name GetWheelStatus
32 * @brief 获取轮子状态以及相关参数
33 * @details
34 *
35 * @param[in] _ucPort 端口号
36 *
37 * @returns ERR_NONE Successfully
38 * Other values Unsuccessfully
39 * @author
40 *********************************************************************************************************
41 */
42 u8 GetWheelStatus(COM_PORT_E _ucPort)
43 {
44  u8 unBuf[256];
45  u8 unLength=0,unID=0;
46  u16 uni,unj,unCheckCRC;
47  UART_T *pUart;
48  uint16_t usCount;
49  pUart = ComToUart(_ucPort);
50  if (NULL==pUart) return 0;
51  /* usRxWrite 变量在中断函数中被改写，主程序读取该变量时，必须进行临界区保护 */
52  DISABLE_INT();
53  usCount = pUart->usRxCount;
54  ENABLE_INT();
55  for(uni=0;uni<3;uni++)
56  {
57  sWheelMotor[uni].nReadSpeedHZ=30001;
58  }
59  if(usCount>256)
60  {
61  usCount=256;
62  }
63  if (usCount< 5) /* 小于6个数 */
64  {
65  return 0;
66  }
67  else
68  {
69  for(uni=0;uni<usCount;uni++)
70  {
71  unBuf[uni]= pUart->pRxBuf[pUart->usRxRead]; /* 从串口接收FIFO取1个数据 */
72  DISABLE_INT();
73  if (++pUart->usRxRead >= pUart->usRxBufSize)
74  {
75  pUart->usRxRead = 0;
76  }
77  pUart->usRxCount--;
78  ENABLE_INT();
79  }
80  for(unj=0;unj<usCount;unj++)
81  {
82  if((unj+5)>usCount)
83  {
84  DISABLE_INT();
85  if(pUart->usRxRead-(usCount-unj)>= 0 )
86  {
87  pUart->usRxRead=pUart->usRxRead-(usCount-unj);
88  }
89  else
90  {
91  pUart->usRxRead=pUart->usRxRead+pUart->usRxBufSize-(usCount-unj);
92  }
93  pUart->usRxCount=pUart->usRxCount+(usCount-unj);
94  ENABLE_INT();
95  return 1;
96  }
97  if((unBuf[unj+1]==0x03)&&(unBuf[unj]==0x01||unBuf[unj]==0x02||unBuf[unj]==0x03))
98  {
99  unID=unBuf[unj];
100  unLength=unBuf[unj+2];
101  if((unLength+unj+5)<=usCount)
102  {
103  unCheckCRC=CRC16_Modbus(&unBuf[unj], unLength+3); //计算CRC校验位
104  if((unLength==0x0E)&&(unCheckCRC==(unBuf[unj+4+unLength]+(unBuf[unj+3+unLength]<<8))))
105  {
106  if(3!=unID)
107  s485WholeCount[unID+14].un485RecieveCount++;
108  else
109  s485WholeCount[14].un485RecieveCount++;
110  sWheelMotor[unID-1].unReadCurrent = (unBuf[unj+3]<<8) + unBuf[unj+4]; /* 当前电流 需要*0.01为实际电流值 */
111  sWheelMotor[unID-1].nReadSpeedHZ = (unBuf[unj+5]<<8) + unBuf[unj+6]; /* 当前速度读回值 除以10为实时换向频率 */
112  sWheelMotor[unID-1].unReadPosArrivalStatus = (unBuf[unj+7]<<8) + unBuf[unj+8]; /*位置是否到达，0未到达，1是到达 */
113  sWheelMotor[unID-1].unReadPositionH = (unBuf[unj+9]<<8) + unBuf[unj+10];
114  sWheelMotor[unID-1].unReadPositionL = (unBuf[unj+11]<<8)+ unBuf[unj+12]; /* 当前位置，换向脉冲数表示， 高低字节拼起来有正负表示方向，一圈1200个脉冲 */
115  sWheelMotor[unID-1].unReadTimeLeftH = (unBuf[unj+13]<<8)+ unBuf[unj+14];
116  sWheelMotor[unID-1].unReadTimeLeftL = (unBuf[unj+15]<<8)+ unBuf[unj+16]; /* 完成设定位置剩余时间 ，高低拼起来单位ms */
117  unj=unj+unLength+5-1;
118  continue;
119  }
120  if((unLength==0x04)&&(unCheckCRC==(unBuf[unj+4+unLength]+(unBuf[unj+3+unLength]<<8))))
121  {
122  sWheelMotor[unID-1].unReadStallStatus = (unBuf[unj+3]<<8) + unBuf[unj+4]; /* 堵转状态 0未堵转 1正转堵转 2反转堵转 */
123  sWheelMotor[unID-1].unReadErrorStatus = (unBuf[unj+5]<<8) + unBuf[unj+6]; /* 错误状态 0无错； 1尚未学习； 2 堵转停止， 3 霍尔错误 ，4 无法到达目标速度 */
124  unj=unj+unLength+5-1;
125  continue;
126  }
127  }
128  else
129  {
130  DISABLE_INT();
131  if(pUart->usRxRead-(usCount-unj)>= 0 )
132  {
133  pUart->usRxRead=pUart->usRxRead-(usCount-unj);
134  }
135  else
136  {
137  pUart->usRxRead=pUart->usRxRead+pUart->usRxBufSize-(usCount-unj);
138  }
139  pUart->usRxCount=pUart->usRxCount+(usCount-unj);
140  ENABLE_INT();
141  return 1;
142  }
143  }
144  else if(((unBuf[unj+1]==0x83)||(unBuf[unj+1]==0x86)||(unBuf[unj+1]==0x90))&&(unBuf[unj]==0x01||unBuf[unj]==0x02))
145  {
146  unID=unBuf[unj];
147  unCheckCRC=CRC16_Modbus(&unBuf[unj], 3); //计算CRC校验位
148  if(unCheckCRC==(unBuf[unj+4]+(unBuf[unj+3]<<8)))
149  {
150  if(unBuf[unj+1]==0x83)
151  {
152  sWheelMotor[unID-1].unReadErrorCode = unBuf[unj+2];
153  }
154  else
155  {
156  sWheelMotor[unID-1].unWriteErrorCode = unBuf[unj+2];
157  }
158  unj=unj+5-1;
159  continue;
160  }
161  }
162  else if(((unBuf[unj+1]==0x06)||(unBuf[unj+1]==0x10))&&(unBuf[unj]==0x01||unBuf[unj]==0x02))
163  {
164  if((unj+8)<=usCount)
165  {
166  unCheckCRC=CRC16_Modbus(&unBuf[unj], 6); //计算CRC校验位
167 
168  if(unCheckCRC==(unBuf[unj+7]+(unBuf[unj+6]<<8)))
169  {
170  unj=unj+8-1;
171  continue;
172  }
173  }
174  else
175  {
176  DISABLE_INT();
177  if(pUart->usRxRead-(usCount-unj)>= 0 )
178  {
179  pUart->usRxRead=pUart->usRxRead-(usCount-unj);
180  }
181  else
182  {
183  pUart->usRxRead=pUart->usRxRead+pUart->usRxBufSize-(usCount-unj);
184  }
185  pUart->usRxCount=pUart->usRxCount+(usCount-unj);
186  ENABLE_INT();
187  return 1;
188  }
189  }
190  }
191  }
192  return 0;
193 }
