257 u8 SendArmDataToExePack(void)

261  sShoulderMotor.nPosCurrent=(sWheelMotor[2].unReadPositionH<<16)+sWheelMotor[2].unReadPositionL;
262  if (sArmMotorCmdPack.unCmbCount != 0) /* Arm已有数据 */
263  {
264  for(uni=0;uni<Arm_Servo_Num+Shoulder_Motor_Num;uni++)
265  {
266  //手臂舵机全部停止
267  if (30000==sArmMotorCmdPack.sCmdBuffer[sArmMotorCmdPack.unCmbRead][uni].nSpeed)
268  {
269  unArmBrakeFlag=1;
270  sArmMotorCmdPack.unCmbRead=0;
271  sArmMotorCmdPack.unCmbCount=0;
272  sArmMotorCmdPack.unCmbWrite=0;
273  InterpolateBrake(ARM_NUM);
274  for(uni=0;uni<Arm_Servo_Num+Shoulder_Motor_Num;uni++)
275  sMyServo[uni].unServoSendID=uni+1;
276  return 0;
277  }
278  unArmBrakeFlag=0;
279  }
280  if (0==unArmStartFlag)
281  {
282  ArmStart(Arm_Start_ID,Arm_Servo_Num+Shoulder_Motor_Num);//手臂运动开始标志.
283  ArmSetCMD(Arm_Start_ID,Arm_Servo_Num+Shoulder_Motor_Num);
284  }else
285  {
286  for (uni = Arm_Start_ID; uni < Arm_Servo_Num+Shoulder_Motor_Num; uni++)
287  {
288  if (0==sMyServo[uni].sControl.unStartFlag)//尝试启动舵机
289  {
290  ArmStart(uni,1);
291  ArmSetCMD(uni,1);
292  }
293  }
294  }
295  if (1==unArmStartFlag&&sArmMotorCmdPack.unCmbCount != 0)
296  GenerateArmGoalPositionSpeed();//生成插值点
297  SendShoulderExePack();//发送肩膀运动数据
298  if (1==unArmInterPolateFlag)//更新执行命令
299  {
300  DISABLE_INT();
301  if (sArmMotorCmdPack.unCmbRead!=sArmMotorCmdPack.unCmbWrite&&sArmMotorCmdPack.unCmbCount>0)
302  sArmMotorCmdPack.unCmbRead++;
303  if (sArmMotorCmdPack.unCmbRead >= ACTION_EXE_PACK_SIZE){
304  sArmMotorCmdPack.unCmbRead = 0;}
305  sArmMotorCmdPack.unCmbCount--;
306  ENABLE_INT();
307  if(sArmMotorCmdPack.unCmbCount!=0)
308  ArmSetCMD(Arm_Start_ID,Arm_Servo_Num+Shoulder_Motor_Num);
309  }
310  }else {
311  if(0==sWheelMotor[2].nReadSpeedHZ)//肩膀停止后，更新
312  {
313  sShoulderMotor.nPosInit=sShoulderMotor.nPosCurrent;
314  sShoulderMotor.nRealPosInit= sShoulderMotor.nRealPosCurrent;
315  }
316  //发送肩膀数据
317  SendShoulderExePack();
318  unArmMotorControlFlag=0;
319  unArmStartFlag=0;
320  if(1==sArmMotorCmdPack.unModeFlag)//若速度模式执行
321  {
322  unArmMotorControlFlag=1;
323  for (uni = Arm_Start_ID; uni < Arm_Servo_Num; uni++)
324  sMyServo[uni].sControl.unBrakeFlag=1;
325  }
326  }
327  /*计算肩膀相对位置*/
328  sShoulderMotor.nDiffPos=sShoulderMotor.nPosCurrent-sShoulderMotor.nPosInit;
329  sShoulderMotor.nDiffPos=sShoulderMotor.nDiffPos*SHOULDER_POS_TO_ROBOT_POS;
330  /*由相对位置计算当前真实位置*/
331  sShoulderMotor.nRealPosCurrent=sShoulderMotor.nRealPosInit+sShoulderMotor.nDiffPos;
332  return 1;
333 }
